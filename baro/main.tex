\begin{frame}{De la physique en plongée}
\only<1>{%
Une pression est une force par surface $P = \frac{F}{S}$:
\begin{itemize}
\item pression en \emph{Pascal} (S.I. [\unit{Pa}] = [\unit{kg \cdot s^{-2} \cdot m^{-1}}]),
\item force en \emph{Newton} (S.I. [\unit{N}] = [\unit{kg \cdot m \cdot s^{-2}}]),
\item surface en \emph{mètre carré} (S.I. [\unit{m^2}].
\end{itemize}
}
